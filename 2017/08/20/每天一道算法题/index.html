我的算法之路

     一天一道PAT

PAT 1004

1. 深度搜索

导语:

在本题中，条件是:给出一些城市，每个城市都有一定数量的救援队，而每个给出城市是是互通的，而我们处于其中一个城市，要到另一已知城市去，现在情况十分紧急，于是问题的中心转化为寻找两点间最短路径。而在本题中只需要说明有几条最短路径，和找出最短路径上的最大搜救队数量。

解题思路

找最短路径有很多方法，其中比较著名的是:Dijkstra和Floyd,而在本题中的解法有点类似于Floyd,从一点出发，不断寻找距离这一点最近的下一个节点，并标注。而再往下找是一个递归过程，直到找出一条最短路径并把路径节点上的每个权值相加即可。

代码如下:

  #include <string.h>
  #include <stdio.h>
  #include <stdlib.h>
  int ncity;
  int nroad;
  int to;
  #define MAX_CITY 505
  const int INF = 0x3f3f3f3f;
  int res_team[MAX_CITY];
  int road[MAX_CITY][MAX_CITY];
  int npath;
  int max_team;
  int lowest_cost;
  int vis[MAX_CITY];
  void dfs(int cur_city, int nteam, int cost){
      int i;
      nteam += res_team[cur_city];
      if(cur_city == to){
          if(cost < lowest_cost){
              npath = 1;
              max_team = nteam;
              lowest_cost = cost;
          }
          else if(cost == lowest_cost){
              npath++;
              if(max_team < nteam){
                  max_team = nteam;
              }
          }
          return ;
      }

      for(i = 0; i < MAX_CITY; i++){
          if(road[cur_city][i] != INF && !vis[i]){
              vis[i] = 1;
              dfs(i, nteam, cost + road[cur_city][i]);
              vis[i]=0;
          }
      }
  }

  int main(){
      int i;
      int from;

      memset(res_team, 0, sizeof(res_team));
      memset(road, 0x3f,sizeof(road));
      memset(vis, 0, sizeof(vis));
      scanf("%d%d%d%d", &ncity, &nroad, &from, &to);

      for(i = 0; i < ncity; i++){
          scanf("%d",&res_team[i]);
      }
      for(i = 0; i < nroad; i++){
          int a, b, cost;
          scanf("%d %d %d",&a, &b, &cost);
          road[a][b] = road[b][a] = cost;
      }
      npath = 0;//走过的路径
      max_team = 0;//最大人数
      vis[from] = 1;//初始化起点城市
      lowest_cost = INF;
      dfs(from,0,0);
      printf("%d %d", npath, max_team);
      return 0;
}
